              def fact(n)
                return 1 if n == 0
                n * fact(n-1)
              end

              puts fact(5)
            

              fn fact(n: u32) -> u32 {
                  match n {
                      0 => 1,
                      i => i * fact(i-1)
                  }
              }

              fn main() {
                  println!("{}", fact(5))
              }
            

              fn fact(n: u32) -> u32 {
                let mut acc = 1;
                for num in 2..=n {
                    acc *= num;
                }
                acc
              }

              fn main() {
                  println!("{}", fact(5))
              }

take a string, return the sum of numbers


              def sum_numbers(s)
                s.split.map(&:to_i).sum
              end
            

              fn sum_numbers(s: &str) -> u32 {
                s.split_whitespace()
                  .map(|x| x.parse::<u32>().unwrap())
                  .sum()
              }
            

              fn sum_numbers(s: &str) -> u32 {
                s.split_whitespace()
                  .map(|x| x.parse::<u32>().expect("number is not an integer"))
                  .sum()
              }
  • ::<u32> - similar to generics in C#
  • unwrap/expect

maybe ignore non-integer numbers?


              fn sum_numbers(s: &str) -> u32 {
                s.split_whitespace()
                  .map(|x| parse_number(x))
                  .sum()
              }

              fn parse_number(x: &str) -> u32 {
                  match x.parse() {
                      Ok(number) => number,
                      Err(_details) => 0 // ignore non-int numbers
                  }
              }
